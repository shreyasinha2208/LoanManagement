1. create table customers(uci number(16) primary key, user_id varchar2(12) unique not null, password varchar2(15) not null, applicant_id number(5) not null,first_name varchar2(20) not null, last_name varchar2(20), father_name varchar2(40) not null, mother_name varchar2(40) not null, dob date not null, email_id varchar2(35) not null, gender varchar2(10) not null, mobile_number varchar2(10) not null, alternate_mobile_number varchar2(10) not null,aadhar_number varchar2(12) not null, pan_number varchar2(10) not null, account_type varchar2(10) not null, account_holder varchar2(10) not null, login_count number);
2. create table loan(loan_number number(6) primary key, uci number(16) not null, loan_amount number(9) not null, loan_tenure number(4) not null, balance number(11,2) not null, applied_date date not null, total_num_of_emis number(4) not null, num_of_emis_paid number(4) not null, type_id number(2) not null, emi_amount number(11,2) not null, applicant_num number(5) foreign key(type_id) references loan_type(type_id) foreign key(applicant_num) references loan_applicant(applicant_num));
3. create table loan_type(type_id number(2) primary key, loan_type varchar2(15) not null, interest_rate varchar(4,2) not null, maximum_limit number(11,2) not null, minimum_limit number(6) not null); 
4. create table emi_payment(trans_id number(10) primary key, loan_number number(6) not null, trans_amount number(20,2) not null, trans_date_time timestamp(6) not null foreign key(loan_number) references loan(loan_number));
5. create table loan_applicant(applicant_num number(5) primary key, uci number(16) not null, document blob not null, status varchar(10) not null);
